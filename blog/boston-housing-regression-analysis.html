Dataset Overview

Economic Context & Historical Background

The Boston Housing dataset was originally collected in the 1970s as part of a study on air pollution and housing values in the Boston metropolitan area. This dataset has become a classic benchmark for regression algorithms and economic analysis.

üìä Dataset Characteristics:

  • Samples: 506 census tracts in Boston metropolitan area
  • Features: 13 socioeconomic and environmental variables
  • Target: Median home value in thousands of dollars (MEDV)
  • Time Period: Early 1970s Boston metropolitan area
  • Problem Type: Continuous value prediction (regression)

Outlier Treatment in Economic Data

Economic data often contains legitimate extreme values that shouldn't be removed carelessly:

# Outlier analysis using IQR method
def detect_outliers_iqr(data, column):
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = data[(data[column] < lower_bound) | (data[column] > upper_bound)]
    return outliers, lower_bound, upper_bound

print("Outlier Analysis for Key Economic Variables:")
print("=" * 50)

# Analyze outliers in target variable
outliers_medv, lower_medv, upper_medv = detect_outliers_iqr(df, 'MEDV')
print(f"MEDV (Housing Prices):")
print(f"  Outlier range: < ${lower_medv:.1f}k or > ${upper_medv:.1f}k")
print(f"  Number of outliers: {len(outliers_medv)} ({len(outliers_medv)/len(df)*100:.1f}%)")

# Analyze outliers in key features
key_features = ['CRIM', 'RM', 'LSTAT', 'DIS']

Feature Engineering for Economic Analysis

Creating Polynomial Features

Economic relationships are often non-linear - let's capture these patterns:

from sklearn.preprocessing import PolynomialFeatures
from sklearn.feature_selection import SelectKBest, f_regression

# Create polynomial features for key economic indicators
key_features_for_poly = ['RM', 'LSTAT', 'DIS', 'NOX']
X_train_poly = X_train_scaled.copy()
X_test_poly = X_test_scaled.copy()

print("Creating Polynomial Features for Economic Relationships:")
print("=" * 60)

for feature in key_features_for_poly:
    # Create squared terms
    X_train_poly[f'{feature}_squared'] = X_train_scaled[feature] ** 2
    X_test_poly[f'{feature}_squared'] = X_test_scaled[feature] ** 2
    
    print(f"‚úÖ Added {feature}_squared")

# Create interaction terms for economically meaningful relationships
interactions = [
    ('RM', 'LSTAT'),    # Rooms vs socioeconomic status
    ('CRIM', 'NOX'),    # Crime vs pollution
    ('TAX', 'PTRATIO'), # Property tax vs education quality
]

for feat1, feat2 in interactions:
    interaction_name = f'{feat1}_{feat2}_interaction'
    X_train_poly[interaction_name] = X_train_scaled[feat1] * X_train_scaled[feat2]
    X_test_poly[interaction_name] = X_test_scaled[feat1] * X_test_scaled[feat2]
    print(f"‚úÖ Added {interaction_name}")

print(f"\nOriginal features: {X_train_scaled.shape[1]}")
print(f"With polynomial features: {X_train_poly.shape[1]}")
print(f"Added {X_train_poly.shape[1] - X_train_scaled.shape[1]} new features")

        'RMSE': round(test_rmse, 3),

    print(f"\nüìä {name}")

    print(f"   Test R¬≤:  {test_r2:.4f}")

# Practical prediction example
def predict_house_price(model, scaler, feature_names, **kwargs):
    """
    Predict house price given property characteristics
    """
    # Create feature vector
    features = []
    feature_dict = {
        'CRIM': kwargs.get('crime_rate', 0.1),
        'ZN': kwargs.get('residential_zoning', 10.0),
        'INDUS': kwargs.get('industrial_proportion', 5.0),
        'CHAS': kwargs.get('river_access', 0),
        'NOX': kwargs.get('air_pollution', 0.5),
        'RM': kwargs.get('rooms', 6.0),
        'AGE': kwargs.get('building_age', 50.0),
        'DIS': kwargs.get('employment_distance', 4.0),
        'RAD': kwargs.get('highway_access', 5),
        'TAX': kwargs.get('property_tax', 300),
        'PTRATIO': kwargs.get('pupil_teacher_ratio', 18.0),
        'B': kwargs.get('demographic_b', 390.0),
        'LSTAT': kwargs.get('lower_status_pct', 10.0)
    }
    
    # Create DataFrame and scale
    input_df = pd.DataFrame([feature_dict])
    input_scaled = scaler.transform(input_df)
    
    # Predict
    prediction = model.predict(input_scaled)[0]
    return prediction
